<Info>
  This page shows you how to connect a user account to Mio, store tokens safely, and bring personalized Mio Context into your product. You will need Mio dashboard access, a redirect URL, Node.js 18+, and a server environment (Next.js route handler, Express API, etc.).
</Info>

## Ship a working flow in six steps

<Steps>
  <Step title="Register your app">
    Visit the <a href="https://dashboard.mio.xyz/en/developer" target="_blank" rel="noreferrer">Mio dashboard</a>, create an OAuth client, and copy the `clientId`, `clientSecret`, and redirect URL you intend to use locally. You need these values before any code changes.
  </Step>

  <Step title="Install the SDK">
    Add the Mio SDK once at the workspace root (monorepos included).

    <CodeGroup>
```bash npm
npm install @mio/mio-sdk
```
```bash yarn
yarn add @mio/mio-sdk
```
```bash pnpm
pnpm add @mio/mio-sdk
```
    </CodeGroup>
  </Step>

  <Step title="Configure environment variables">
    Provide Mio credentials on both the browser and the server. Prefix client-side values with `NEXT_PUBLIC_` (or your framework equivalent).

    <Snippet src="/snippets/env-variables" />

    <Warning>
      Never expose your Mio client secret environment variable to the browser. Keep it next to the server-side Mio instance so tokens are exchanged safely.
    </Warning>
  </Step>

  <Step title="Create the token exchange endpoint">
    Initialize Mio on the server once, then reuse the singleton inside the route that swaps authorization codes for tokens.

<Tabs>
<Tab title="Next.js route handler">

```typescript app/api/exchange-token/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { Mio } from '@mio/mio-sdk/server';

const mio = Mio.init({
  clientId: process.env.MIO_CLIENT_ID!,
  redirectUrl: process.env.MIO_REDIRECT_URL!,
  clientSecret: process.env.MIO_CLIENT_SECRET!
});

export async function POST(request: NextRequest) {
  const { code } = await request.json();
  const tokens = await mio.exchangeCodeForTokens(code);

  // Persist refreshToken before returning it to the client
  // await saveTokensForUser(userId, tokens);

  return NextResponse.json(tokens);
}
```

</Tab>
<Tab title="Express endpoint">

```typescript src/routes/mio.ts
import express from 'express';
import { Mio } from '@mio/mio-sdk/server';

const router = express.Router();
const mio = Mio.init({
  clientId: process.env.MIO_CLIENT_ID!,
  redirectUrl: process.env.MIO_REDIRECT_URL!,
  clientSecret: process.env.MIO_CLIENT_SECRET!
});

router.post('/api/exchange-token', async (req, res) => {
  const tokens = await mio.exchangeCodeForTokens(req.body.code);
  res.json(tokens);
});

export default router;
```

</Tab>
</Tabs>

    <Tip>
      Store `refreshToken` alongside your user record so you can call `mio.refreshTokens` later without asking the user to reconnect.
    </Tip>

    <Warning>
      Ensure the redirect URL in your code matches the dashboard configuration exactly; otherwise the exchange fails with `invalid_grant`.
    </Warning>
  </Step>

  <Step title="Initialize Mio in your frontend">
    Initialize Mio once in your frontend so you can trigger the connection flow and later fetch Mio Context, whether you use React or not.

<Tabs>
<Tab title="React (with MioProvider)">

```typescript app/providers.tsx
'use client';

import type { ReactNode } from 'react';
import { MioProvider } from '@mio/mio-sdk/react';

const mioConfig = {
  clientId: process.env.NEXT_PUBLIC_MIO_CLIENT_ID!,
  redirectUrl: `${process.env.NEXT_PUBLIC_APP_URL}/api/mio/callback`,
  exchangeTokenUrl: '/api/exchange-token'
};

export function Providers({ children }: { children: ReactNode }) {
  return <MioProvider config={mioConfig}>{children}</MioProvider>;
}
```

</Tab>
<Tab title="Vanilla HTML + JavaScript">

```html index.html
<button id="mio-connect">Connect with Mio</button>

<script type="module">
  import { Mio } from '@mio/mio-sdk/client';

  const mio = Mio.init({
    clientId: 'YOUR_MIO_CLIENT_ID',
    redirectUrl: `${window.location.origin}/mio/callback`,
    exchangeTokenUrl: '/api/exchange-token'
  });

  document
    .getElementById('mio-connect')
    ?.addEventListener('click', () => {
      mio.connect();
    });
</script>
```

</Tab>
</Tabs>

    <Tip>
      In React, `MioProvider` memoizes the underlying SDK. In non-React apps, keep the `Mio.init` call in a shared module so you reuse the same instance everywhere.
    </Tip>
  </Step>

  <Step title="Fetch Mio Context & personalize your experience">
    Use Mio helpers to manage the OAuth flow, capture tokens, and request Mio Context (via `chat` or `getSummary`) from either React or plain JavaScript.

<Tabs>
<Tab title="React">

```typescript app/page.tsx
'use client';

import { useState } from 'react';
import { useMio } from '@mio/mio-sdk/react';

export default function HomePage() {
  const { connect, handleMioCallback, chat, getSummary, isLoading, error } = useMio();
  const [accessToken, setAccessToken] = useState<string | null>(null);
  const [answer, setAnswer] = useState<string | null>(null);

  async function handleConnect() {
    if (!accessToken) {
      await connect();
    }
  }

  async function completeCallback() {
    const tokens = await handleMioCallback();
    setAccessToken(tokens.accessToken);
  }

  async function askMio() {
    if (!accessToken) {
      throw new Error('Connect first so we can get an access token');
    }
    const response = await chat({ query: 'Summarize the user', accessToken });
    setAnswer(response);
  }

  async function loadSummary() {
    if (!accessToken) {
      throw new Error('Connect first so we can get an access token');
    }
    const summary = await getSummary({ accessToken });
    setAnswer(summary ?? 'No summary returned');
  }

  return (
    <section>
      <button onClick={handleConnect} disabled={isLoading}>
        {isLoading ? 'Connectingâ€¦' : 'Connect with Mio'}
      </button>
      <button onClick={completeCallback}>Handle callback</button>
      <button onClick={askMio}>Get Mio Context</button>
      <button onClick={loadSummary}>Load user summary</button>
      {error ? <p>{error}</p> : null}
      {answer ? <pre>{answer}</pre> : null}
    </section>
  );
}
```

</Tab>
<Tab title="Vanilla HTML + JavaScript">

```html callback.html
<section>
  <button id="mio-complete">Finish Mio connection</button>
  <button id="mio-context">Get Mio Context</button>
  <button id="mio-summary">Get user summary</button>
</section>

<script type="module">
  import { Mio } from '@mio/mio-sdk/client';

  const mio = Mio.init({
    clientId: 'YOUR_MIO_CLIENT_ID',
    redirectUrl: `${window.location.origin}/mio/callback`,
    exchangeTokenUrl: '/api/exchange-token'
  });

  let accessToken = null;

  async function completeCallback() {
    const params = new URLSearchParams(window.location.search);
    const code = params.get('code');
    if (!code) return;

    const tokens = await mio.exchangeCode(code);
    accessToken = tokens.accessToken;

    // Clean up the URL once the code has been consumed
    window.history.replaceState({}, document.title, window.location.pathname);
  }

  async function getMioContext() {
    if (!accessToken) {
      throw new Error('Connect first so we can get an access token');
    }
    const response = await mio.chat({
      query: 'Summarize the user for a welcome email',
      accessToken
    });
    console.log('Mio Context', response);
  }

  async function getSummary() {
    if (!accessToken) {
      throw new Error('Connect first so we can get an access token');
    }
    const summary = await mio.getUserSummary({ accessToken });
    console.log('Mio user summary', summary);
  }

  document
    .getElementById('mio-complete')
    ?.addEventListener('click', () => {
      void completeCallback();
    });

  document
    .getElementById('mio-context')
    ?.addEventListener('click', () => {
      void getMioContext();
    });

  document
    .getElementById('mio-summary')
    ?.addEventListener('click', () => {
      void getSummary();
    });
</script>
```

## What just happened

- You registered a Mio OAuth client and installed the SDK.
- You configured environment variables and stood up a secure token exchange route.
- You wrapped your React tree with `MioProvider`, handled the callback, and called both `chat` and `getSummary` using a real access token.
